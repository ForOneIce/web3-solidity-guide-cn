核心概念详解

🔌 模块化设计 - 插件架构

将功能分离到独立的插件合约中,核心合约动态调用:

架构设计:
┌──────────────────────────────────┐
│    PluginStore (核心合约)         │
│  - 管理玩家配置文件                │
│  - 注册/管理插件地址               │
│  - 动态调用插件功能                │
└──────────┬───────────────────────┘
           │
    ┌──────┴──────┐
    │             │
┌───▼────┐   ┌───▼────┐
│ 成就插件│   │武器插件│
│Achievements  │WeaponStore
└────────┘   └────────┘

优势:

  • ✅ 关注点分离: 核心逻辑和扩展功能分开
  • ✅ 易于扩展: 添加新功能只需部署新插件
  • ✅ 独立升级: 可以单独升级某个插件
  • ✅ 降低复杂度: 每个合约职责单一
📞 三种低级调用方法
调用类型 作用 上下文 用途
call 调用其他合约 被调用合约 修改其他合约状态
staticcall 只读调用 被调用合约 查询数据(view)
delegatecall 执行其他代码 调用者合约 代理模式/升级
🔄 call - 跨合约调用
// 调用插件的setAchievement函数
bytes memory data = abi.encodeWithSignature(
    "setAchievement(address,string)", 
    user, 
    "First Victory"
);

(bool success, ) = pluginAddress.call(data);
require(success, "Call failed");

执行上下文:

PluginStore.runPlugin()
    │
    │ call →
    │
    ▼
AchievementsPlugin.setAchievement()
    │
    └─ 修改 AchievementsPlugin 的存储
    └─ msg.sender = PluginStore地址
    └─ 在 AchievementsPlugin 的上下文中执行

特点:

  • 修改被调用合约的状态
  • msg.sender变为调用者地址
  • 返回布尔值+返回数据
  • 失败不会自动revert,需要手动检查
🔍 staticcall - 只读调用
// 查询插件数据(不修改状态)
bytes memory data = abi.encodeWithSignature(
    "getAchievement(address)", 
    user
);

(bool success, bytes memory result) = pluginAddress.staticcall(data);
require(success, "Call failed");

string memory achievement = abi.decode(result, (string));

与call的区别:

  • ✅ 只能调用view或pure函数
  • ✅ 不能修改任何状态
  • ✅ Gas消耗更低
  • ✅ 如果被调用函数试图修改状态,会自动revert
✅ 最佳实践: 查询数据时使用staticcall,修改数据时使用call
⚡ delegatecall - 在调用者上下文执行

这是最特殊的调用方式,代码在被调用合约中,但执行环境在调用者中:

// 使用另一个合约的代码,但修改自己的存储
bytes memory data = abi.encodeWithSignature("someFunction()");
(bool success, ) = logicContract.delegatecall(data);

执行上下文对比:

call:
ContractA.call(ContractB.func())
    └─ 执行ContractB的代码
    └─ 修改ContractB的存储
    └─ msg.sender = ContractA

delegatecall:
ContractA.delegatecall(ContractB.func())
    └─ 执行ContractB的代码
    └─ 修改ContractA的存储  ← 关键区别!
    └─ msg.sender = 原始调用者
⚠️ 重要:

delegatecall非常强大但危险!使用时必须确保两个合约的存储布局完全一致,否则会覆盖错误的存储槽位!

🔐 ABI编码与解码

低级调用需要手动编码函数调用数据:

// 编码: 创建调用数据
bytes memory data = abi.encodeWithSignature(
    "functionName(type1,type2)",  // 函数签名
    arg1,                          // 参数1
    arg2                           // 参数2
);

// 或者使用函数选择器
bytes4 selector = bytes4(keccak256("functionName(type1,type2)"));
bytes memory data = abi.encodeWithSelector(selector, arg1, arg2);

// 解码: 解析返回数据
(bool success, bytes memory returnData) = target.call(data);
(uint256 result, string memory message) = abi.decode(returnData, (uint256, string));

编码函数:

  • abi.encode(...): 完整ABI编码
  • abi.encodePacked(...): 紧凑编码(用于哈希)
  • abi.encodeWithSignature(...): 带函数签名编码
  • abi.encodeWithSelector(...): 带选择器编码
  • abi.decode(...): 解码数据

完整代码详解

📄 核心合约: PluginStore

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract PluginStore {
    // 玩家配置文件
    struct PlayerProfile {
        string name;
        string avatar;
    }
    
    // 状态变量
    mapping(address => PlayerProfile) public profiles;
    mapping(string => address) public plugins;
    
    // 设置配置文件
    function setProfile(string memory _name, string memory _avatar) external {
        profiles[msg.sender] = PlayerProfile({
            name: _name,
            avatar: _avatar
        });
    }
    
    // 获取配置文件
    function getProfile(address user) external view returns (string memory, string memory) {
        PlayerProfile memory profile = profiles[user];
        return (profile.name, profile.avatar);
    }
    
    // 注册插件
    function registerPlugin(string memory key, address pluginAddress) external {
        plugins[key] = pluginAddress;
    }
    
    // 获取插件地址
    function getPlugin(string memory key) external view returns (address) {
        return plugins[key];
    }
    
    // 运行插件(修改状态) - 使用call
    function runPlugin(
        string memory key,
        string memory functionSignature,
        address user,
        string memory argument
    ) external {
        address plugin = plugins[key];
        require(plugin != address(0), "Plugin not found");
        
        // 编码函数调用
        bytes memory data = abi.encodeWithSignature(
            functionSignature, 
            user, 
            argument
        );
        
        // 调用插件
        (bool success, ) = plugin.call(data);
        require(success, "Plugin call failed");
    }
    
    // 查询插件(只读) - 使用staticcall
    function runPluginView(
        string memory key,
        string memory functionSignature,
        address user
    ) external view returns (string memory) {
        address plugin = plugins[key];
        require(plugin != address(0), "Plugin not found");
        
        // 编码函数调用
        bytes memory data = abi.encodeWithSignature(functionSignature, user);
        
        // 只读调用
        (bool success, bytes memory result) = plugin.staticcall(data);
        require(success, "Plugin call failed");
        
        // 解码返回数据
        return abi.decode(result, (string));
    }
}

📄 插件1: 成就系统

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract AchievementsPlugin {
    // 存储每个玩家的最新成就
    mapping(address => string) public latestAchievement;
    
    // 设置成就
    function setAchievement(address user, string memory achievement) public {
        latestAchievement[user] = achievement;
    }
    
    // 获取成就
    function getAchievement(address user) public view returns (string memory) {
        return latestAchievement[user];
    }
}

✅ 架构优势

PluginStore管理核心数据(玩家配置文件),各个插件管理各自的功能数据(成就、武器等)。添加新功能只需部署新插件并注册,无需修改核心合约!

🔍 完整调用流程解析

场景: 玩家获得成就

// 步骤1: 部署并注册插件
AchievementsPlugin achievementsPlugin = new AchievementsPlugin();
pluginStore.registerPlugin("achievements", address(achievementsPlugin));

// 步骤2: 玩家调用PluginStore
pluginStore.runPlugin(
    "achievements",                           // 插件key
    "setAchievement(address,string)",        // 函数签名
    playerAddress,                            // 玩家地址
    "Defeated Dragon"                         // 成就名称
);

// 内部执行流程:
1. PluginStore查找plugins["achievements"]
2. 获取插件地址: 0xAchievementsPlugin...
3. 编码调用数据:
   data = abi.encodeWithSignature(
       "setAchievement(address,string)",
       playerAddress,
       "Defeated Dragon"
   )
4. 执行call:
   (success, ) = 0xAchievementsPlugin.call(data)
5. AchievementsPlugin执行setAchievement:
   latestAchievement[playerAddress] = "Defeated Dragon"

// 步骤3: 查询成就
string memory achievement = pluginStore.runPluginView(
    "achievements",
    "getAchievement(address)",
    playerAddress
);
// 返回: "Defeated Dragon"

函数签名生成规则

// 规则: "functionName(type1,type2,type3)"
// 注意: 类型之间没有空格!

// ✅ 正确
"setAchievement(address,string)"
"transfer(address,uint256)"
"swap(address,address,uint256)"

// ❌ 错误
"setAchievement(address, string)"   // 有空格
"transfer(address,uint)"            // uint不完整,应该是uint256
"swap(address[], uint)"             // 数组类型要写完整

// 特殊类型
"func(uint256[])"      // 动态数组
"func(uint256[5])"     // 固定大小数组
"func(bytes32)"        // bytes32
"func(string)"         // string
"func(address[])"      // 地址数组

⚔️ call vs delegatecall 实战对比

call 示例

contract A {
    uint public value;
    
    function callB(address b) external {
        b.call(
            abi.encodeWithSignature(
                "setValue(uint256)", 
                42
            )
        );
    }
}

contract B {
    uint public value;
    
    function setValue(uint x) public {
        value = x;  // 修改B的value
    }
}

// 结果:
// A.value = 0 (未改变)
// B.value = 42 (被修改)

delegatecall 示例

contract A {
    uint public value;
    
    function delegatecallB(address b) external {
        b.delegatecall(
            abi.encodeWithSignature(
                "setValue(uint256)", 
                42
            )
        );
    }
}

contract B {
    uint public value;
    
    function setValue(uint x) public {
        value = x;  // 修改A的value!
    }
}

// 结果:
// A.value = 42 (被修改!)
// B.value = 0 (未改变)

📚 深入理解

为什么delegatecall危险?

// 存储布局必须一致!
contract Proxy {
    address public implementation;  // 槽位0
    uint256 public value;           // 槽位1
}

contract Logic {
    uint256 public value;           // 槽位0 ⚠️
    address public something;       // 槽位1 ⚠️
}

// 如果Proxy.delegatecall(Logic),
// Logic的value会写入Proxy的槽位0,
// 覆盖implementation地址! 💥

真实世界应用

  • Uniswap: 使用delegatecall实现可升级合约
  • Aave: 模块化架构,各功能独立合约
  • Compound: 通过call调用价格预言机
  • Axie Infinity: 游戏逻辑分离到不同合约

函数选择器计算

    "transfer(address,uint256)", 
    recipient, 
    amount
);
